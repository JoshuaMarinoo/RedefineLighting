const feed   = document.getElementById('feed');

const canvas = document.getElementById('canvas');

const ctx    = canvas.getContext('2d');


window.latestStatus = null;


function resizeCanvas() {

  const r  = feed.getBoundingClientRect();

  const wr = document.getElementById('videoWrap').getBoundingClientRect();


  canvas.style.left = (r.left - wr.left) + 'px';

  canvas.style.top  = (r.top  - wr.top)  + 'px';

  canvas.width  = r.width;

  canvas.height = r.height;


  if (window.latestStatus) {

    drawIphoneBox(window.latestStatus);

  }

}


feed.addEventListener('load', resizeCanvas);

window.addEventListener('resize', resizeCanvas);

resizeCanvas();


function drawIphoneBox(status) {

  ctx.clearRect(0, 0, canvas.width, canvas.height);


  if (!status || !status.iphone_box || canvas.width === 0 || canvas.height === 0) {

    return;

  }


  const box = status.iphone_box;


  const x = (box.x - box.width / 2) * canvas.width;

  const y = (box.y - box.height / 2) * canvas.height;

  const w = box.width * canvas.width;

  const h = box.height * canvas.height;


  const cx = x + w / 2;

  const cy = y + h / 2;


  ctx.fillStyle = 'rgba(0, 230, 140, 0.08)';

  ctx.fillRect(x, y, w, h);


  ctx.strokeStyle = '#00e68c';

  ctx.lineWidth = 3;

  ctx.setLineDash([]);

  ctx.strokeRect(x, y, w, h);


  ctx.fillStyle = '#00e68c';

  ctx.beginPath();

  ctx.arc(cx, cy, 4, 0, Math.PI * 2);

  ctx.fill();


  ctx.font = '12px DM Mono, monospace';

  ctx.fillStyle = '#00e68c';

  const conf = typeof box.confidence === 'number' ? box.confidence.toFixed(2) : '—';

  ctx.fillText(`conf ${conf}`, x + 6, Math.max(14, y - 6));

}


async function post(url, body={}) {

  const r = await fetch(url, {

    method:'POST',

    headers:{'Content-Type':'application/json'},

    body:JSON.stringify(body)

  });


  return r.json();

}


async function cmdReset() {

  await post('/api/reset');

  window.latestStatus = null;

  ctx.clearRect(0, 0, canvas.width, canvas.height);

}


async function cmdScanPorts() {

  const sel = document.getElementById('portSelect');


  sel.innerHTML = '<option value="">Scanning…</option>';

  document.getElementById('btnConnect').disabled = true;


  const res = await fetch('/api/servo/list_ports').then(r => r.json());


  sel.innerHTML = '';


  if (!res.ok || !res.ports.length) {

    sel.innerHTML = '<option value="">No ports found</option>';

    return;

  }


  res.ports.forEach(p => {

    const opt = document.createElement('option');

    opt.value = p.device;

    opt.textContent = `${p.device}  —  ${p.description}`;


    if (p.device === res.connected) {

      opt.selected = true;

    }


    sel.appendChild(opt);

  });


  document.getElementById('btnConnect').disabled = false;

}


async function cmdConnect() {

  const port = document.getElementById('portSelect').value;


  if (!port) {

    return;

  }


  const btn = document.getElementById('btnConnect');


  btn.disabled = true;

  btn.innerHTML = '<span class="btn-icon">…</span> Connecting';


  const res = await post('/api/servo/connect', { port });


  btn.innerHTML = '<span class="btn-icon">⚡</span> Connect';


  if (res.ok) {

    showServoControls(true);

    document.getElementById('connectedBadge').textContent = `● Connected  ${port}`;

  } else {

    alert('Connection failed: ' + (res.error || 'unknown error'));

    btn.disabled = false;

  }

}


async function cmdDisconnect() {

  await post('/api/servo/disconnect');

  showServoControls(false);

}


function showServoControls(connected) {

  document.getElementById('servoConnect').style.display  = connected ? 'none' : '';

  document.getElementById('servoControls').style.display = connected ? '' : 'none';

}


const LED_STEPS = [0, 25, 50, 75, 100];

let ledDebounce = null;


function updateLed(idx) {

  const pct = LED_STEPS[parseInt(idx)];


  document.getElementById('ledVal').textContent = pct + '%';


  clearTimeout(ledDebounce);


  ledDebounce = setTimeout(() => {

    post('/api/led', { brightness: pct });

  }, 150);

}


function syncLedSlider(pct) {

  const idx = LED_STEPS.indexOf(pct);


  if (idx !== -1) {

    document.getElementById('ledSlider').value = idx;

  }


  document.getElementById('ledVal').textContent = pct + '%';

}

}


async function cmdServoHome() {

}

}

function updateAngleIndicator(deg, minDeg=-10, maxDeg=10) {

  const fill = document.getElementById('angleFill');

  const pct  = (deg - minDeg) / (maxDeg - minDeg);


  document.getElementById('angleLabel').textContent = deg.toFixed(1) + '°';


  if (deg >= 0) {

    fill.style.left = '50%';

    fill.style.width = (pct - 0.5) * 100 + '%';

  } else {

    fill.style.left = pct * 100 + '%';

    fill.style.width = (0.5 - pct) * 100 + '%';

  }

}


let settingsSynced = false;


async function poll() {

  try {

    const s = await fetch('/api/status').then(r => r.json());


    window.latestStatus = s;


    const dot = document.getElementById('statusDot');

    dot.className = 'status-dot';


    if (s.iphone_connected) {

      dot.classList.add('tracking');

    } else if (s.tracking_lost) {

      dot.classList.add('warn');

    } else if (s.frame_w > 0) {

      dot.classList.add('live');

    }


    document.getElementById('statusText').textContent = s.status;


    const iphoneConnected = !!s.iphone_connected;

    const motorMovement = s.iphone_motor_movement || (s.servo_enabled ? 'ON' : 'PAUSED');

    const gCommand = s.iphone_g_command || '—';

    const handGesture = s.iphone_hand_gesture || 'none';

    const box = s.iphone_box;


    document.getElementById('statIphone').textContent = iphoneConnected ? 'YES' : 'NO';

    document.getElementById('statIphone').className =

      'stat-value' + (iphoneConnected ? '' : ' warn');


    document.getElementById('statState').textContent = iphoneConnected ? 'IPHONE' : 'WAIT';

    document.getElementById('statState').className =

      'stat-value' + (iphoneConnected ? '' : ' dim');


    document.getElementById('statG').textContent = gCommand;

    document.getElementById('statG').className =

      'stat-value' + (gCommand === '—' ? ' dim' : '');


    document.getElementById('statMotor').textContent = motorMovement;

    document.getElementById('statMotor').className =

      'stat-value' + (motorMovement === 'PAUSED' ? ' warn' : '');


    document.getElementById('statHand').textContent = handGesture;

    document.getElementById('statHand').className =

      'stat-value' + (handGesture === 'none' ? ' dim' : '');


    document.getElementById('statConf').textContent =

      box && typeof box.confidence === 'number' ? box.confidence.toFixed(2) : '—';

    document.getElementById('statConf').className =

      'stat-value' + (box ? '' : ' dim');


    document.getElementById('statW').textContent = s.frame_w || '—';

    document.getElementById('statH').textContent = s.frame_h || '—';


    showServoControls(s.servo_connected);

    updateServoButton(s.servo_enabled);

    updateAngleIndicator(s.servo_angle || 0);


    if (typeof s.led_brightness !== 'undefined') {

      syncLedSlider(s.led_brightness);

    }


    if (!settingsSynced && s.gain) {

      document.getElementById('gainSlider').value = s.gain;

      document.getElementById('gainVal').textContent = s.gain.toFixed(2);

      document.getElementById('deadzoneSlider').value = s.deadzone;

      document.getElementById('deadzoneVal').textContent = s.deadzone;

      settingsSynced = true;

    }


    drawIphoneBox(s);

  } catch(e) {

    console.error(e);

  }

}


setInterval(poll, 400);

cmdScanPorts();

poll();
